44  double updatePointCloud(const std::vector<util::Point> &x,
45  bool parallel = true) {
46 
47  auto t1 = steady_clock::now();
48  if (parallel) {
49  d_cloud_p->points.resize(x.size());
50 
51  hpx::parallel::for_loop(
52  hpx::parallel::execution::par, 0, x.size(),
53  [this, &x](boost::uint64_t i) {
54  (*this->d_cloud_p)[i].x = x[i].d_x;
55  (*this->d_cloud_p)[i].y = x[i].d_y;
56  (*this->d_cloud_p)[i].z = x[i].d_z;
57  });
58  } else {
59  for (size_t i=0; i<x.size(); i++) {
60  (*this->d_cloud_p)[i].x = x[i].d_x;
61  (*this->d_cloud_p)[i].y = x[i].d_y;
62  (*this->d_cloud_p)[i].z = x[i].d_z;
63  }
64  }
65  auto t2 = steady_clock::now();
66  return util::methods::timeDiff(t1, t2);
67  }
68 
69  virtual double setInputCloud() = 0;
70 
71 public:
72  /*@brief control the verbosity */
73  size_t d_debug;
74  /*@brief name of tree: kdtree or octree */
75  std::string d_treeType;
76  /*@brief coordinates of the points */
77  pcl::PointCloud<pcl::PointXYZ>::Ptr d_cloud_p;
