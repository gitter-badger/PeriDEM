237  RnpMaterial(inp::MaterialDeck &deck, const size_t &dim, const double &horizon)
238  : Material("RNPBond"), d_horizon(horizon), d_density(deck.d_density),
239  d_C(0.), d_beta(0.), d_rbar(0.), d_invFactor(0.),
240  d_factorSc(deck.d_checkScFactor),
241  d_irrevBondBreak(deck.d_irreversibleBondBreak) {
242 
243  // set global fields
244  if (dimension != dim)
245  dimension = dim;
246 
247  if (is_plane_strain != deck.d_isPlaneStrain)
248  is_plane_strain = deck.d_isPlaneStrain;
249 
250  // create influence function
251  if (deck.d_influenceFnType == 0) {
252  if (influence_fn == nullptr)
253  influence_fn = std::make_shared<material::ConstInfluenceFn>(
254  deck.d_influenceFnParams, dim);
255  }
256  else if (deck.d_influenceFnType == 1) {
257  if (influence_fn == nullptr)
258  influence_fn = std::make_shared<material::LinearInfluenceFn>(
259  deck.d_influenceFnParams, dim);
260  }
261  else if (deck.d_influenceFnType == 2) {
262  if (influence_fn == nullptr)
263  influence_fn = std::make_shared<material::GaussianInfluenceFn>(
264  deck.d_influenceFnParams, dim);
265  }
266  else {
267  std::cerr << "Error: Influence function type = "
268  << deck.d_influenceFnType
269  << " is invalid.\n";
270  exit(1);
271  }
272 
273  if (dim == 1)
274  d_invFactor = std::pow(horizon, 2) * 2.;
275  else if (dim == 2)
276  d_invFactor = std::pow(horizon, 3) * M_PI;
277  else if (dim == 3)
278  d_invFactor = std::pow(horizon, 4) * 4. * M_PI / 3.;
279 
280  // check if we need to compute the material parameters
281  if (deck.d_computeParamsFromElastic)
282  computeParameters(deck, dim);
283  else {
284  d_C = deck.d_bondPotentialParams[0];
285  d_beta = deck.d_bondPotentialParams[1];
286  d_rbar = std::sqrt(0.5 / d_beta);
287  }
288  };

321  std::pair<double, double> getBondEF(const double &r, const double &s,
322  bool &fs,
323  const bool &break_bonds) const override {
324 
325  if (break_bonds) {
326  // check if fracture state of the bond need to be updated
327  if (d_irrevBondBreak && !fs &&
328  util::isGreater(std::abs(s), d_factorSc * getSc(r)))
329  fs = true;
330 
331  // if bond is not fractured, return energy and force from nonlinear
332  // potential otherwise return energy of fractured bond, and zero force
333  if (!fs)
334  return std::make_pair(
335  getInfFn(r) * d_C *
336  (1. - std::exp(-d_beta * r * s * s)) / d_invFactor,
337  getInfFn(r) * 4. * s * d_C * d_beta *
338  std::exp(-d_beta * r * s * s) / d_invFactor);
339  else
340  return std::make_pair(d_C / d_invFactor, 0.);
341  } else {
342  return std::make_pair(getInfFn(r) * d_C * d_beta *
343  r * s * s / d_invFactor,
344  getInfFn(r) * 4. * s * d_C *
345  d_beta / d_invFactor);
346  }
347  };

434  inp::MatData computeMaterialProperties(const size_t &dim) const override {
435 
436  auto data = inp::MatData();
437 
438  // set Poisson's ratio to 1/4
439  data.d_nu = 0.25;
440

444  // compute peridynamic parameters

448  } else if (dim == 3) {

451  }

453  data.d_G = data.d_lambda;
454  data.d_E = data.toELambda(data.d_lambda,
455  data.d_nu);
456  data.d_K =
457  data.toK(data.d_E, data.d_nu);
458  data.d_KIc = data.toKIc(
459  data.d_Gc, data.d_nu, data.d_E);

495 private:
496  void computeParameters(inp::MaterialDeck &deck, const size_t &dim) {
497  //
498  // Need following elastic and fracture properties
499  // 1. E or K
500  // 2. Gc or KIc
501  // For bond-based, Poisson's ratio is fixed to 1/4
502  //
503  if (util::isLess(deck.d_matData.d_E, 0.) &&
504  util::isLess(deck.d_matData.d_K, 0.)) {
505  std::cerr << "Error: Require either Young's modulus E or Bulk modulus K"
506  " to compute the RNP bond-based peridynamic parameters.\n";
507  exit(1);
508  }
509  if (util::isGreater(deck.d_matData.d_E, 0.) &&
510  util::isGreater(deck.d_matData.d_K, 0.)) {
511  std::cout << "Warning: Both Young's modulus E and Bulk modulus K are "
512  "provided.\n";
513  std::cout << "Warning: To compute the RNP bond-based peridynamic "
514  "parameters, we only require one of those.\n";
515  std::cout
516  << "Warning: Selecting Young's modulus to compute parameters.\n";
517  }
518 
519  if (util::isLess(deck.d_matData.d_Gc, 0.) &&
520  util::isLess(deck.d_matData.d_KIc, 0.)) {
521  std::cerr << "Error: Require either critical energy release rate Gc or "
522  "critical stress intensity factor KIc to compute the RNP "
523  "bond-based peridynamic parameters.\n";
524  exit(1);
525  } else if (util::isGreater(deck.d_matData.d_Gc, 0.) &&
526  util::isGreater(deck.d_matData.d_KIc, 0.)) {
527  std::cout << "Warning: Both critical energy release rate Gc and critical "
528  "stress intensity factor KIc are provided.\n";
529  std::cout << "Warning: To compute the RNP bond-based peridynamic "
530  "parameters, we only require one of those.\n";
531  std::cout << "Warning: Selecting critical energy release rate Gc to "
532  "compute parameters.\n";
533  }
534 
535  // set Poisson's ratio to 1/4
536  deck.d_matData.d_nu = 0.25;
537 
538  // compute E if not provided or K if not provided
539  if (deck.d_matData.d_E > 0.)
540  deck.d_matData.d_K =
541  deck.d_matData.toK(deck.d_matData.d_E, deck.d_matData.d_nu);
542 
543  if (deck.d_matData.d_K > 0. && deck.d_matData.d_E < 0.)
544  deck.d_matData.d_E =
545  deck.d_matData.toE(deck.d_matData.d_K, deck.d_matData.d_nu);
546 
547  if (deck.d_matData.d_Gc > 0.)
548  deck.d_matData.d_KIc = deck.d_matData.toKIc(
549  deck.d_matData.d_Gc, deck.d_matData.d_nu, deck.d_matData.d_E);
550 
551  if (deck.d_matData.d_KIc > 0. && deck.d_matData.d_Gc < 0.)
552  deck.d_matData.d_Gc = deck.d_matData.toGc(
553  deck.d_matData.d_KIc, deck.d_matData.d_nu, deck.d_matData.d_E);
554 
555  // compute lame parameter
556  deck.d_matData.d_lambda =
557  deck.d_matData.toLambdaE(deck.d_matData.d_E, deck.d_matData.d_nu);
558  deck.d_matData.d_G =
559  deck.d_matData.toGE(deck.d_matData.d_E, deck.d_matData.d_nu);
560  deck.d_matData.d_mu = deck.d_matData.d_G;
561 
562  // get moment of influence function
563  double M = getMoment(dim);
564 
565  // compute peridynamic parameters
566  if (dim == 2) {
567  d_C = M_PI * deck.d_matData.d_Gc / (4. * M);
568  d_beta = 4. * deck.d_matData.d_lambda / (d_C * M);
569  } else if (dim == 3) {
570  d_C = 2. * deck.d_matData.d_Gc / (3. * M);
571  d_beta = 5. * deck.d_matData.d_lambda / (d_C * M);
572  }
573 
574  d_rbar = std::sqrt(0.5 / d_beta);
575  };
